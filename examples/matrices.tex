% Matrices
% More information here: https://www.overleaf.com/learn/latex/Matrices

% Packages necessary
\usepackage{amsmath}



\begin{document}

% Plain; no delimiters
\begin{matrix}
    a & b \\
    c & d
\end{matrix}

% Matrix with parentheses
\begin{pmatrix}
    a & b \\
    c & d
\end{pmatrix}

% Matrix with square brackets; brackets
\begin{bmatrix}
    a & b \\
    c & d
\end{bmatrix}

% Matrix with curly brackets; braces
\begin{Bmatrix}
    a & b \\
    c & d
\end{Bmatrix}

\end{document}
