% Equations

\usepackage{amsmath}
\usepackage{physics}


% Normal equation environment
\begin{equation}
    2 + 2 = 4
\end{equation}

% Use the align environment for multi-line equations
% The "&" marks where to align each line (you are in the "align" environment, hint, hint)
\begin{align}
    a & = \frac{b}{c} \times \frac{d}{e} \\
    & = \frac{bd}{ce}
\end{align}

% Environments with asterisks (*) result in unnumbered environments. This may be an ideal option.

% the physics package provides easy use of derivatives and related symbols. 
% No need to use methods like \frac{\mathbb{d}y}{\mathbb{d}x} 
\begin{equation*}
    \int x\,\dd{x}
\end{equation*}
\begin{equation*}
    \dv{y}{x}
\end{equation*}
